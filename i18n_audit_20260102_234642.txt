     6	export type LanguageContextType = {
     7	  language: string;
     8	  setLanguage: (lang: string) => void;
     9	  t: (key: string) => string;
    10	};
    11	
    12	const LanguageContext = createContext<LanguageContextType | null>(null);
    13	
    14	export function LanguageProvider({ children }: { children: React.ReactNode }) {
    15	  const [language, setLanguage] = useState<string>("es");
    16	
    17	  const value = useMemo<LanguageContextType>(() => {
    18	    const defaultLanguage = "es";
    19	
    20	    const t = (key: string) => {
    21	      const currentDict = translations[language] || translations[defaultLanguage] || {};
    22	      const fallbackDict = translations[defaultLanguage] || {};
    23	      return currentDict[key] ?? fallbackDict[key] ?? key;
    24	    };
    25	
    26	    return { language, setLanguage, t };
    27	  }, [language]);
    28	
    29	  return <LanguageContext.Provider value={value}>{children}</LanguageContext.Provider>;
    30	}
    31	
    32	export function useLanguage() {
    33	  const ctx = useContext(LanguageContext);
    34	  if (!ctx) throw new Error("useLanguage debe usarse dentro de <LanguageProvider>");

    36	}
